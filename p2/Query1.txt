SELECT C_NAME, O_ORDERKEY, SUM(LINEITEM.L_QUANTITY * (LINEITEM.L_EXTENDEDPRICE - LINEITEM.L_DISCOUNT)) AS "Order Size"
FROM ORDERS
INNER JOIN LINEITEM ON LINEITEM.L_ORDERKEY = ORDERS.O_ORDERKEY
INNER JOIN CUSTOMER ON CUSTOMER.C_CUSTKEY = ORDERS.O_CUSTKEY
GROUP BY O_ORDERKEY
ORDER BY "Order Size" DESC
LIMIT 20;
