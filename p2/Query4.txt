SELECT S_NAME, COUNT(DISTINCT C_CUSTKEY) AS total
FROM SUPPLIER
INNER JOIN LINEITEM ON LINEITEM.L_SUPPKEY = SUPPLIER.S_SUPPKEY
INNER JOIN ORDERS ON LINEITEM.L_ORDERKEY = ORDERS.O_ORDERKEY
INNER JOIN CUSTOMER ON ORDERS.O_CUSTKEY = CUSTOMER.C_CUSTKEY
GROUP BY S_SUPPKEY
/* filter by distinct C_CUSTKEY */
HAVING total > 615
ORDER BY total DESC, S_NAME;
